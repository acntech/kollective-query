        $having:MIN(employees.salary)$gte:10000 =>

        Example 2:

        $having:COUNT(employees)$gte:10 =>

        SELECT d FROM Department d WHERE d.id IN ( SELECT e.department.id FROM Employee e GROUP BY e.department.id HAVING COUNT(e) >= 10000 )
